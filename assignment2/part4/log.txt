To run: 

setup ssh tunnel in virtual box

make sure camera is enabled

setup camera remap using ffmpeg: 

sudo modprobe v4l2loopback devices=1 exclusive_caps=1

ffmpeg -f v4l2              \
       -input_format mjpeg  \
       -framerate 1        \
       -video_size 640x360 \
       -i /dev/video0       \
       -pix_fmt yuyv422     \
       -f v4l2 /dev/video2  

v4l2-ctl --list-devices

insert kernel module:
sh krun.sh

then start server
sh server.sh 

then start client   
sh client.sh
